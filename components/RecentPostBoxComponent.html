import { Component, ChangeDetectionStrategy, Input } from '@angular/core';
import { EMPTY_RECENT_POST } from '@shared/data/data';
import { Post } from '@shared/types/interface.post';

@Component({
  selector: 'app-recent-post-box',
  templateUrl: './recent-post-box.component.html',
  styleUrls: ['./recent-post-box.component.scss'],
  changeDetection: ChangeDetectionStrategy.Default
})

export class RecentPostBoxComponent {

  @Input() posts: Post[] = EMPTY_RECENT_POST;
  @Input() margin: boolean | undefined;
  @Input() limit: number = 3;

  constructor() { }

}

@defer(when posts) {
  <div class="recent" [ngClass]="{'margin': margin}">
    <ul>
      @for(post of posts; track post._id) {
        <li [routerLink]="'/post/' + post.slug">
          <div class="wrapper">
            <img [src]="post.cover" alt="Post Image" 
            #img
            (error)="img.src = 'assets/images/placeholder.png'; img.alt='No Image'; img.className='no-image'"/>
          </div>
          <div class="info">
            <h1>{{post.title}}</h1>
            <span>{{post.author}}</span>
            <span class="date">{{post.createdAt | date}}</span>
          </div>
        </li>
      }
    </ul>
  </div>
} @placeholder {
  <app-recent-post-skeleton [limit]="limit"/>
}
